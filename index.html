Create a free Supabase project
Supabase stores all survey responses and teams in a free hosted database accessible from any device or browser.
  1. Go to supabase.com → Start your project
  2. Sign up with GitHub (fastest) and create a new project
  3. Pick any region, set a database password, wait ~2 min
  4. Once the project is ready, click Next
Create the database tables
In Supabase go to SQL Editor and run this query. It creates the teams and responses tables.
-- Teams table
create table if not exists teams (
  id uuid primary key default gen_random_uuid(),
  name text not null unique,
  created_at timestamptz default now()
);

-- Responses table
create table if not exists responses (
  id uuid primary key default gen_random_uuid(),
  team_id uuid references teams(id) on delete cascade,
  team_name text not null,
  answers jsonb not null,
  submitted_at timestamptz default now()
);

-- Row-level security (allow public read/write — no login needed for respondents)
alter table teams enable row level security;
alter table responses enable row level security;

create policy "public read teams"    on teams     for select using (true);
create policy "public insert teams"  on teams     for insert with check (true);
create policy "public delete teams"  on teams     for delete using (true);
create policy "public read resp"     on responses for select using (true);
create policy "public insert resp"   on responses for insert with check (true);
create policy "public delete resp"   on responses for delete using (true);
